### arr.reduce
1. implement .map as .reduce
```js
function map(arr, func) {
  return arr.reduce(
    (acc, item) => {
      acc.push(func(item));
      return acc;
    }, [] //[]는 초기값 세팅
  )
}
```
2. implement .filter as .reduce
```js
function filter(arr, func) {
  return arr.reduce(
    (acc, item) => { 
      func(item) ? acc.push(item) : null
      return acc;
    }, [] //[]는 초기값 세팅
  )
}
```
